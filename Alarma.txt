class Alarma expands Eventos{
    public Alarma(Prioridad prioridad, date fecha, time horaLimite){
        this.prioridad = prioridad;
        this.fecha = fecha;
        this.horaLimite = horaLimite;
        this.id = classID++;
    }
    //Este metodo se llamar√≠a cada minuto para comprobar la hora
    public void bloquearSistema(){
        if(horaLimite == time.now()){
            main::alarmaForm.show();
        }
    }
}
